    def to_blob
      blob = []
      blob <<  [137, 80, 78, 71, 13, 10, 26, 10].pack("C*")
      blob << PNG.chunk('IHDR', [@width, @height, BITS, RGB, NONE, NONE, NONE].pack("N2C5"))
      blob << PNG.chunk('IDAT', Zlib::Deflate.deflate(self.png_join))
      blob << PNG.chunk('IEND', '')
      blob.join
    end
