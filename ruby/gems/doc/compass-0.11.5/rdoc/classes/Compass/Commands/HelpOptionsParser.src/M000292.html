      def command_list(header, commands)
        list = "#{header}\n"
        commands.sort_by{|c| c.to_s}.each do |command|
          list << "  * #{command}"
          if Compass::Commands[command].respond_to? :description
            list << "\t- #{Compass::Commands[command].description(command)}"
          end
          list << "\n"
        end
        list
      end
