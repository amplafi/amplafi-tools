        def parse!(arguments)
          parser = option_parser(arguments)
          parser.parse!
          if arguments.size == 0
            raise OptionParser::ParseError, "Please specify the grid dimensions."
          end
          parser.options[:grid_dimensions] = arguments.shift
          parser.options[:grid_filename] = arguments.shift
          parser.options
        end
