        def parse_init!(arguments)
          parser = option_parser(arguments)
          parse_options!(parser, arguments)
          if arguments.size > 0
            parser.options[:project_type] = arguments.shift.to_sym
          end
          parse_arguments!(parser, arguments)
          set_default_arguments(parser)
          parser.options
        end
