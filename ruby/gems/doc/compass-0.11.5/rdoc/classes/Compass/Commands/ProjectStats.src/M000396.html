      def css_columns(css_file)
        if File.exists?(css_file)
          cf = Compass::Stats::CssFile.new(css_file)
          cf.analyze!
          %w(selector_count prop_count).map do |t|
            cf.send(t).to_s
          end
        else
          return [ '--', '--' ]
        end
      rescue LoadError
        @missing_css_parser = true
        return [ 'DISABLED', 'DISABLED' ]
      end
