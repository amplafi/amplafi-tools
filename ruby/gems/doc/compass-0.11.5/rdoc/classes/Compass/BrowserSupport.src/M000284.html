    def remove_support(function, *aspects)
      aspects.each do |aspect|
        unless ASPECTS.include?(aspect)
          Compass::Util.compass_warn "Unknown support aspect: #{aspect}"
          next
        end
        SIMPLE_FUNCTIONS[function.to_s].reject!{|a| a == aspect.to_s}
      end
    end
