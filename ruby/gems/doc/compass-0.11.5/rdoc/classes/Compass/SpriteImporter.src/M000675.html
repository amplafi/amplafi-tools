    def content_for_images(skip_overrides = false)
      "@import \"compass/utilities/sprites/base\";\n\n// General Sprite Defaults\n// You can override them before you import this file.\n$\#{name}-sprite-base-class: \".\#{name}-sprite\" !default;\n$\#{name}-sprite-dimensions: false !default;\n$\#{name}-position: 0% !default;\n$\#{name}-spacing: 0 !default;\n$\#{name}-repeat: no-repeat !default;\n$\#{name}-prefix: '' !default;\n$\#{name}-clean-up: true !default;\n\n\#{skip_overrides ? \"$\#{name}-sprites: sprite-map(\\\"\#{uri}\\\", $cleanup: $\#{name}-clean-up);\" : generate_overrides }\n\n// All sprites should extend this class\n// The \#{name}-sprite mixin will do so for you.\n\\\#{$\#{name}-sprite-base-class} {\nbackground: $\#{name}-sprites no-repeat;\n}\n\n// Use this to set the dimensions of an element\n// based on the size of the original image.\n@mixin \#{name}-sprite-dimensions($name) {\n@include sprite-dimensions($\#{name}-sprites, $name)\n}\n\n// Move the background position to display the sprite.\n@mixin \#{name}-sprite-position($name, $offset-x: 0, $offset-y: 0) {\n@include sprite-background-position($\#{name}-sprites, $name, $offset-x, $offset-y)\n}\n\n// Extends the sprite base class and set the background position for the desired sprite.\n// It will also apply the image dimensions if $dimensions is true.\n@mixin \#{name}-sprite($name, $dimensions: $\#{name}-sprite-dimensions, $offset-x: 0, $offset-y: 0) {\n@extend \\\#{$\#{name}-sprite-base-class};\n@include sprite($\#{name}-sprites, $name, $dimensions, $offset-x, $offset-y)\n}\n\n@mixin \#{name}-sprites($sprite-names, $dimensions: $\#{name}-sprite-dimensions, $prefix: sprite-map-name($\#{name}-sprites)) {\n@include sprites($\#{name}-sprites, $sprite-names, $\#{name}-sprite-base-class, $dimensions, $prefix)\n}\n\n// Generates a class for each sprited image.\n@mixin all-\#{name}-sprites($dimensions: $\#{name}-sprite-dimensions, $prefix: sprite-map-name($\#{name}-sprites)) {\n@include \#{name}-sprites(\#{sprite_names.join(\" \")}, $dimensions, $prefix);\n}\n"
    end
