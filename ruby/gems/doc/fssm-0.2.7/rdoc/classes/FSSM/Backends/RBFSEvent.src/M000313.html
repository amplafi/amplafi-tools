    def run
      begin
        @fsevent = FSEvent.new
        @fsevent.watch(temporary_multipath_hack) do |paths|
          paths.each do |path|
            temporary_multipath_handler(path)
          end
        end
        @fsevent.run
      rescue Interrupt
        @fsevent.stop
      end
    end
