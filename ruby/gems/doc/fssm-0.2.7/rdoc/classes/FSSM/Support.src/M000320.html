    def usable_backend
      choice = case
                 when mac? && !lion? && !jruby? && carbon_core?
                   'FSEvents'
                 when mac? && rb_fsevent?
                   'RBFSEvent'
                 when linux? && rb_inotify?
                   'Inotify'
                 else
                   'Polling'
               end

      if (mac? || linux?) && choice == 'Polling'
        optimal = case
                    when mac?
                      'rb-fsevent'
                    when linux?
                      'rb-inotify'
                  end
        FSSM.dbg("An optimized backend is available for this platform!")
        FSSM.dbg("    gem install #{optimal}")
      end

      choice
    end
