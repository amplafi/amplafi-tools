    def to_sass(opts = {})
      precedence = Sass::Script::Parser.precedence_of(separator)
      value.map do |v|
        if v.is_a?(List) && Sass::Script::Parser.precedence_of(v.separator) <= precedence
          "(#{v.to_sass(opts)})"
        else
          v.to_sass(opts)
        end
      end.join(sep_str(nil))
    end
