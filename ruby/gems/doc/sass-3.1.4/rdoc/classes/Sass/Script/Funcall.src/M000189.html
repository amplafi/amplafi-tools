      def inspect
        args = @args.map {|a| a.inspect}.join(', ')
        keywords = @keywords.sort_by {|k, v| k}.
            map {|k, v| "$#{k}: #{v.inspect}"}.join(', ')
        "#{name}(#{args}#{', ' unless args.empty? || keywords.empty?}#{keywords})"
      end
