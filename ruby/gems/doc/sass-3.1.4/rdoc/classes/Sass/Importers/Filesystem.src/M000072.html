      def find_real_file(dir, name)
        for (f,s) in possible_files(remove_root(name))
          path = (dir == ".") ? f : "#{dir}/#{f}"
          if full_path = Dir[path].first
            full_path.gsub!(REDUNDANT_DIRECTORY,File::SEPARATOR)
            return full_path, s
          end
        end
        nil
      end
